  2 /**@file This is the javascript for the etch-a-sketch game.
  3 * The four arrow keys cause the drawer to move, leaving a "sketch" behind it.
  4 * The space bar starts the game. The letter r resets the canvas to blank.
  5 * The user plays until he/she wants to stop (presses r).*/
  6 
  7 /**Game canvas (square the game is displayed on on the HTML page)*/
  8 var canvas = document.getElementById("myCanvas3");
  9 var ctx = canvas.getContext("2d");
 10 var space = false;
 11 
 12 var spaceStop = false;
 13 var downPressed = false;
 14 var upPressed = false;
 15 var rightPressed = false;
 16 var leftPressed = false;
 17 var Reset = false;
 18 var Erase = false;
 19 
 20 var snakeButtSize = 5;
 21 var numOfSnakeButts = 1;
 22 var SnakeButts = [];
 23 //var Pixels = [];
 24 var Score = 0;
 25 var START = true;
 26 var movetimer = 0;
 27 SnakeButts[0] = { x: 20, y: 20, added: true }
 28 var Pixels = new Array()
 29 for (i = 0; i < 88; i++)
 30     Pixels[i] = new Array()
 31 
 32 for (i = 0; i < 88; i++) {
 33     for (j = 0; j < 88; j++) {
 34         Pixels[j][i] = {x:5*j+20, y:5*i+20, showPixel : false}
 35     }  
 36 }

 38 /**Listening for user key strokes*/
 39 document.addEventListener("keydown", keyDownHandler, false);
 40 document.addEventListener("keyup", keyUpHandler, false);
 41 
 42 /**See which, if any, keys have been pressed.
 43 * Only dealing with arrow keys, space bar, and the letter r.*/
 44 function keyDownHandler(e) {
 45     if (e.keyCode == 38) {
 46         upPressed = true;
 47     }
 48     else if (e.keyCode == 40) {
 49         downPressed = true;
 50     }
 51     else if (e.keyCode == 32) {
 52         space = true;
 53         spaceStop = true;
 54     }
 55     else if (e.keyCode == 39) {
 56         rightPressed = true;
 57     }
 58     else if (e.keyCode == 37) {
 59         leftPressed = true;
 60     }
 61     if (e.keyCode == 82) {
 62         Reset = true;
 63     }
 64     if (e.keyCode == 69) {
 65         Erase = true;
 66     }
 67 }
 68 
 69 /**See which, if any, keys have been released.
 70 * Only dealing with arrow keys, space bar, and the letter r.*/
 71 function keyUpHandler(e) {
 72     if (e.keyCode == 38) {
 73         upPressed = false;
 74     }
 75     else if (e.keyCode == 40) {
 76         downPressed = false;
 77     }
 78     else if (e.keyCode == 32) {
 79         space = false;
 80     }
 81     else if (e.keyCode == 39) {
 82         rightPressed = false;
 83     }
 84     else if (e.keyCode == 37) {
 85         leftPressed = false;
 86     }
 87     if (e.keyCode == 82) {
 88         Reset = false;
 89     }
 90     if (e.keyCode == 69) {
 91         Erase = false;
 92     }
 93 }
 94 
 95 /**When the space bar is pressed, start the game (allow user to move drawer).*/
 96 function spaceBar() {
 97     if (spaceStop == false) {
 98         setTimeout(spaceBar, 50);
 99         clearInterval(timer1)
100         return;
101     }
102     START = false;
103     timer1 = setInterval(draw, 40);
104 }
105 
106 /**when the game is lost, wait 50ms instead of 10ms and then check for space bar, 
107 when space is pressed load function GameOver*/
108 function finalWait() {
109     if (spaceStop == false) {
110         setTimeout(finalWait, 50);
111         clearInterval(timer1)
112         return;
113     }
114     timer1 = setInterval(GameOver, 10);
115 }
116 
117 /**When the game is over, reload the game canvas.*/
118 function GameOver() {
119     document.location.reload();
120 }

123 /**Drawing the "sketch" following the user's drawer? NEED ZACH*/
124 function cursor() {
125     
126     ctx.beginPath();
127     ctx.rect(SnakeButts[0].x, SnakeButts[0].y, snakeButtSize, snakeButtSize);
128     ctx.fillStyle = "#000000";
129     ctx.fill();
130     ctx.closePath();
131     
132 }
133 
134 /**Checks all the 5x5 pixels in 2D array Pixels and if true shows them on the canvas */
135 function drawSketch() {
136 
137     
138     for (k = 0; k < 88; k++) {
139         for (l = 0; l < 88; l++) {
140             if (Pixels[l][k].showPixel == true) {
141                 ctx.beginPath();
142                 ctx.rect(Pixels[l][k].x, Pixels[l][k].y, 5, 5);
143                 ctx.fillStyle = "#696969";
144                 ctx.fill();
145                 ctx.closePath();
146             }
147 
148         }
149     }
150     
151 }

153 /**Refresh the game canvas, called when the user presses 'r'.*/
154 function refresh() {
155     for (k = 0; k < 88; k++) {
156         for (l = 0; l < 88; l++) {
157             Pixels[l][k].showPixel = false;
158 
159         }
160     }
161 }

163 /**Main function that controls canvas visuals. 
164 * Shows the Starting instructions and the controls. 
165 * Refreshes page if Reset == true. 
166 * Looks for the direction variables and moves 5 pixels accordingly.
167 * Initiates the cursor and drawSketch functions. 
168 * Takes the position of the cursor and makes the corresponding 5x5 pixel .showPixel == true*/
169 function draw() {
170     ctx.clearRect(0, 0, canvas.width, canvas.height)
171     if (START == true) {
172         ctx.font = "55px Impact";
173         ctx.fillText("Start Game", 480 / 2 - 125, 480 / 2);
174         ctx.font = "20px Impact";
175         ctx.fillText("by pressing SPACEBAR", 480 / 2 - 90, 480 / 2 + 30);
176     }
177     
178     if (Reset == true){
179         refresh();
180     }
181     if (spaceStop == false) {
182         spaceBar();
183     }
184     
185     
186     drawSketch();
187     cursor();
188     if (downPressed && SnakeButts[0].y < canvas.height - snakeButtSize - 20) {
189         SnakeButts[0].y += 5;
190     }
191     else if (upPressed && SnakeButts[0].y > 20) {
192         SnakeButts[0].y -= 5;
193     }
194     if (rightPressed && SnakeButts[0].x < canvas.width - snakeButtSize - 20) {
195         SnakeButts[0].x += 5;
196     }
197     else if (leftPressed && SnakeButts[0].x > 20) {
198         SnakeButts[0].x -= 5;
199     }
200     var newX = (SnakeButts[0].x - 20) / 5;
201     var newY = (SnakeButts[0].y - 20) / 5;
202     if (Erase == true) {
203         Pixels[newX][newY].showPixel = false;
204     }
205     else {
206         Pixels[newX][newY].showPixel = true;
207     }
208     
209 
210     
211     
212 }
213 //** Re-runs the function draw() every 40 ms*/
214 var timer1 = setInterval(draw, 40);
